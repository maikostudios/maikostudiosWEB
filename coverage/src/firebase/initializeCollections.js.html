/**
 * Crea una colecci√≥n con un documento de ejemplo si no existe
 * @param {string} collectionName - Nombre de la colecci√≥n
 * @param {Object} exampleDoc - Documento de ejemplo con id y data
 */
const createCollectionIfNotExists = async (collectionName, exampleDoc) => {
  try {
    const docRef = doc(db, collectionName, exampleDoc.id);
    const docSnap = await getDoc(docRef);

    if (!docSnap.exists()) {
      await setDoc(docRef, exampleDoc.data);
      console.log(
        `üìÑ Colecci√≥n '${collectionName}' creada con documento de ejemplo`
      );
    } else {
      console.log(`‚úÖ Colecci√≥n '${collectionName}' ya existe`);
    }
  } catch (error) {
    console.error(`‚ùå Error creando colecci√≥n '${collectionName}':`, error);
    throw error;
  }
};
